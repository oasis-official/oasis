import java.util.Map;

import com.dongkuk.oasis.TypedObject;
import com.dongkuk.oasis.context.DefaultServiceContext;
import com.dongkuk.oasis.context.ServiceContext;
import com.dongkuk.oasis.factories.NonTransactionalServiceStarterFactory;
import com.dongkuk.oasis.service.ServiceStarter;

public class Main {

	public static void main(String[] args) {
		ServiceStarter serviceStarter = 
				new NonTransactionalServiceStarterFactory().generateServiceStarter();

		// Build input values
		Map<String, TypedObject> input = new HashMap<>();
		input.put("name", new TypedObject("dongkuk"));

		// Make service context
		ServiceContext serviceContext = new DefaultServiceContext(input);

		serviceStarter.start("my-first-oasis", serviceContext);
	}
}
